public void Configure(EntityTypeBuilder<Client> builder)
        {
            builder.HasOne(c => c.Conseiller)
                .WithMany(l => l.Clients)
                .HasForeignKey(c => c.ConseillerFK)
                .OnDelete(DeleteBehavior.Cascade);
